20 // The gradient checker adds a L2 normalization loss function on top of the
21 // top blobs, and checks the gradient.
22 template <typename Dtype>
23 class GradientChecker {
24  public:
25  // kink and kink_range specify an ignored nonsmooth region of the form
26  // kink - kink_range <= |feature value| <= kink + kink_range,
27  // which accounts for all nonsmoothness in use by caffe
28  GradientChecker(const Dtype stepsize, const Dtype threshold,
29  const unsigned int seed = 1701, const Dtype kink = 0.,
30  const Dtype kink_range = -1)
31  : stepsize_(stepsize), threshold_(threshold), seed_(seed),
32  kink_(kink), kink_range_(kink_range) {}
33  // Checks the gradient of a layer, with provided bottom layers and top
34  // layers.
35  // Note that after the gradient check, we do not guarantee that the data
36  // stored in the layer parameters and the blobs are unchanged.
37  void CheckGradient(Layer<Dtype>* layer, vector<Blob<Dtype>*>* bottom,
38  vector<Blob<Dtype>*>* top, int check_bottom = -1) {
39  layer->SetUp(*bottom, top);
40  CheckGradientSingle(layer, bottom, top, check_bottom, -1, -1);
41  }
42  void CheckGradientExhaustive(Layer<Dtype>* layer,
43  vector<Blob<Dtype>*>* bottom, vector<Blob<Dtype>*>* top,
44  int check_bottom = -1);
45 
46  // CheckGradientEltwise can be used to test layers that perform element-wise
47  // computation only (e.g., neuron layers) -- where (d y_i) / (d x_j) = 0 when
48  // i != j.
49  void CheckGradientEltwise(Layer<Dtype>* layer,
50  vector<Blob<Dtype>*>* bottom, vector<Blob<Dtype>*>* top);
51 
52  void CheckGradientSingle(Layer<Dtype>* layer, vector<Blob<Dtype>*>* bottom,
53  vector<Blob<Dtype>*>* top, int check_bottom, int top_id,
54  int top_data_id, bool element_wise = false);

56  // Checks the gradient of a network. This network should not have any data
57  // layers or loss layers, since the function does not explicitly deal with
58  // such cases yet. All input blobs and parameter blobs are going to be
59  // checked, layer-by-layer to avoid numerical problems to accumulate.
60  void CheckGradientNet(const Net<Dtype>& net,
61  const vector<Blob<Dtype>*>& input);

74 template <typename Dtype>
75 void GradientChecker<Dtype>::CheckGradientSingle(Layer<Dtype>* layer,
76  vector<Blob<Dtype>*>* bottom, vector<Blob<Dtype>*>* top,
77  int check_bottom, int top_id, int top_data_id, bool element_wise) {
78  if (element_wise) {
79  CHECK_EQ(0, layer->blobs().size());
80  CHECK_LE(0, top_id);
81  CHECK_LE(0, top_data_id);
82  const int top_count = (*top)[top_id]->count();
83  for (int blob_id = 0; blob_id < bottom->size(); ++blob_id) {
84  CHECK_EQ(top_count, (*bottom)[blob_id]->count());
85  }
86  }
87  // First, figure out what blobs we need to check against.
88  vector<Blob<Dtype>*> blobs_to_check;
89  for (int i = 0; i < layer->blobs().size(); ++i) {
90  blobs_to_check.push_back(layer->blobs()[i].get());
91  }
92  if (check_bottom < 0) {
93  for (int i = 0; i < bottom->size(); ++i) {
94  blobs_to_check.push_back((*bottom)[i]);
95  }
96  } else {
97  CHECK(check_bottom < bottom->size());
98  blobs_to_check.push_back((*bottom)[check_bottom]);
99  }
100  // Compute the gradient analytically using Backward
101  Caffe::set_random_seed(seed_);
102  // Get any loss from the layer
103  Dtype computed_objective = layer->Forward(*bottom, top);
104  // Get additional loss from the objective
105  computed_objective += GetObjAndGradient(top, top_id, top_data_id);
106  layer->Backward(*top, true, bottom);
107  // Store computed gradients for all checked blobs
108  vector<shared_ptr<Blob<Dtype> > >
109  computed_gradient_blobs(blobs_to_check.size());
110  for (int blob_id = 0; blob_id < blobs_to_check.size(); ++blob_id) {
111  Blob<Dtype>* current_blob = blobs_to_check[blob_id];
112  computed_gradient_blobs[blob_id].reset(new Blob<Dtype>());
113  computed_gradient_blobs[blob_id]->ReshapeLike(*current_blob);
114  const int count = blobs_to_check[blob_id]->count();
115  const Dtype* diff = blobs_to_check[blob_id]->cpu_diff();
116  Dtype* computed_gradients =
117  computed_gradient_blobs[blob_id]->mutable_cpu_data();
118  caffe_copy(count, diff, computed_gradients);
119  }
120  // Compute derivative of top w.r.t. each bottom and parameter input using
121  // finite differencing.
122  // LOG(ERROR) << "Checking " << blobs_to_check.size() << " blobs.";
123  for (int blob_id = 0; blob_id < blobs_to_check.size(); ++blob_id) {
124  Blob<Dtype>* current_blob = blobs_to_check[blob_id];
125  const Dtype* computed_gradients =
126  computed_gradient_blobs[blob_id]->cpu_data();
127  // LOG(ERROR) << "Blob " << blob_id << ": checking "
128  // << current_blob->count() << " parameters.";
129  for (int feat_id = 0; feat_id < current_blob->count(); ++feat_id) {
130  // For an element-wise layer, we only need to do finite differencing to
131  // compute the derivative of (*top)[top_id][top_data_id] w.r.t.
132  // (*bottom)[blob_id][i] only for i == top_data_id. For any other
133  // i != top_data_id, we know the derivative is 0 by definition, and simply
134  // check that that's true.
135  Dtype estimated_gradient = 0;
136  if (!element_wise || (feat_id == top_data_id)) {
137  // Do finite differencing.
138  // Compute loss with stepsize_ added to input.
139  current_blob->mutable_cpu_data()[feat_id] += stepsize_;
140  Caffe::set_random_seed(seed_);
141  Dtype positive_objective = layer->Forward(*bottom, top);
142  positive_objective += GetObjAndGradient(top, top_id, top_data_id);
143  // Compute loss with stepsize_ subtracted from input.
144  current_blob->mutable_cpu_data()[feat_id] -= stepsize_ * 2;
145  Caffe::set_random_seed(seed_);
146  Dtype negative_objective = layer->Forward(*bottom, top);
147  negative_objective += GetObjAndGradient(top, top_id, top_data_id);
148  // Recover original input value.
149  current_blob->mutable_cpu_data()[feat_id] += stepsize_;
150  estimated_gradient = (positive_objective - negative_objective) /
151  stepsize_ / 2.;
152  }
153  Dtype computed_gradient = computed_gradients[feat_id];
154  Dtype feature = current_blob->cpu_data()[feat_id];
155  // LOG(ERROR) << "debug: " << current_blob->cpu_data()[feat_id] << " "
156  // << current_blob->cpu_diff()[feat_id];
157  if (kink_ - kink_range_ > fabs(feature)
158  || fabs(feature) > kink_ + kink_range_) {
159  // We check relative accuracy, but for too small values, we threshold
160  // the scale factor by 1.
161  Dtype scale = max(
162  max(fabs(computed_gradient), fabs(estimated_gradient)), 1.);
163  EXPECT_NEAR(computed_gradient, estimated_gradient, threshold_ * scale)
164  << "debug: (top_id, top_data_id, blob_id, feat_id)="
165  << top_id << "," << top_data_id << "," << blob_id << "," << feat_id;
166  }
167  // LOG(ERROR) << "Feature: " << current_blob->cpu_data()[feat_id];
168  // LOG(ERROR) << "computed gradient: " << computed_gradient
169  // << " estimated_gradient: " << estimated_gradient;
170  }
171  }
172 }

175 void GradientChecker<Dtype>::CheckGradientExhaustive(Layer<Dtype>* layer,
176  vector<Blob<Dtype>*>* bottom, vector<Blob<Dtype>*>* top, int check_bottom) {
177  layer->SetUp(*bottom, top);
178  CHECK_GT(top->size(), 0) << "Exhaustive mode requires at least one top blob.";
179  // LOG(ERROR) << "Exhaustive Mode.";
180  for (int i = 0; i < top->size(); ++i) {
181  // LOG(ERROR) << "Exhaustive: blob " << i << " size " << top[i]->count();
182  for (int j = 0; j < (*top)[i]->count(); ++j) {
183  // LOG(ERROR) << "Exhaustive: blob " << i << " data " << j;
184  CheckGradientSingle(layer, bottom, top, check_bottom, i, j);
185  }
186  }

203 template <typename Dtype>
204 void GradientChecker<Dtype>::CheckGradientNet(
205  const Net<Dtype>& net, const vector<Blob<Dtype>*>& input) {
206  const vector<shared_ptr<Layer<Dtype> > >& layers = net.layers();
207  vector<vector<Blob<Dtype>*> >& bottom_vecs = net.bottom_vecs();
208  vector<vector<Blob<Dtype>*> >& top_vecs = net.top_vecs();
209  for (int i = 0; i < layers.size(); ++i) {
210  net.Forward(input);
211  LOG(ERROR) << "Checking gradient for " << layers[i]->layer_param().name();
212  CheckGradientExhaustive(*(layers[i].get()), bottom_vecs[i], top_vecs[i]);
213  }

216 template <typename Dtype>
217 Dtype GradientChecker<Dtype>::GetObjAndGradient(vector<Blob<Dtype>*>* top,
218  int top_id, int top_data_id) {
219  Dtype loss = 0;
220  if (top_id < 0) {
221  // the loss will be half of the sum of squares of all outputs
222  for (int i = 0; i < top->size(); ++i) {
223  Blob<Dtype>* top_blob = (*top)[i];
224  const Dtype* top_blob_data = top_blob->cpu_data();
225  Dtype* top_blob_diff = top_blob->mutable_cpu_diff();
226  int count = top_blob->count();
227  for (int j = 0; j < count; ++j) {
228  loss += top_blob_data[j] * top_blob_data[j];
229  }
230  // set the diff: simply the data.
231  memcpy(top_blob_diff, top_blob_data, sizeof(Dtype) * top_blob->count());
232  }
233  loss /= 2.;
234  } else {
235  // the loss will be the top_data_id-th element in the top_id-th blob.
236  for (int i = 0; i < top->size(); ++i) {
237  Blob<Dtype>* top_blob = (*top)[i];
238  Dtype* top_blob_diff = top_blob->mutable_cpu_diff();
239  memset(top_blob_diff, 0, sizeof(Dtype) * top_blob->count());
240  }
241  loss = (*top)[top_id]->cpu_data()[top_data_id];
242  (*top)[top_id]->mutable_cpu_diff()[top_data_id] = 1.;
243  }
244  return loss;
245 }
