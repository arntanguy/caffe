174 // The following two macros are modifications of DEFINE_VSL_UNARY_FUNC
175 // in include/caffe/util/mkl_alternate.hpp authored by @Rowland Depp.
176 // Please refer to commit 7e8ef25c7 of the boost-eigen branch.
177 // Git cherry picking that commit caused a conflict hard to resolve and

180 #define DEFINE_CAFFE_CPU_UNARY_FUNC(name, operation) \
181  template<typename Dtype> \
182  void caffe_cpu_##name(const int n, const Dtype* x, Dtype* y) { \
183  CHECK_GT(n, 0); CHECK(x); CHECK(y); \
184  for (int i = 0; i < n; ++i) { \
185  operation; \
186  } \
187  }

189 #define INSTANTIATE_CAFFE_CPU_UNARY_FUNC(name) \
190  template <> \
191  void caffe_cpu_##name<float>(const int n, const float* x, float* y); \
192  template <> \
193  void caffe_cpu_##name<double>(const int n, const double* x, double* y)

196 #define DEFINE_AND_INSTANTIATE_GPU_UNARY_FUNC(name, operation) \
197 template<typename Dtype> \
198 __global__ void name##_kernel(const int n, const Dtype* x, Dtype* y) { \
199  CUDA_KERNEL_LOOP(index, n) { \
200  operation; \
201  } \
202 } \
203 template <> \
204 void caffe_gpu_##name<float>(const int n, const float* x, float* y) { \
205  /* NOLINT_NEXT_LINE(whitespace/operators) */ \
206  name##_kernel<float><<<CAFFE_GET_BLOCKS(n), CAFFE_CUDA_NUM_THREADS>>>( \
207  n, x, y); \
208 } \
209 template <> \
210 void caffe_gpu_##name<double>(const int n, const double* x, double* y) { \
211  /* NOLINT_NEXT_LINE(whitespace/operators) */ \
212  name##_kernel<double><<<CAFFE_GET_BLOCKS(n), CAFFE_CUDA_NUM_THREADS>>>( \
213  n, x, y); \
